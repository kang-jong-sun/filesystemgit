
```python
ema7 = close.ewm(span=7, adjust=False).mean()
ema100 = close.ewm(span=100, adjust=False).mean()

# ADX 계산 (Wilder's Smoothing 필수)
adx = adx_wilder(high, low, close, period=14)

# RSI 계산 (Wilder's Smoothing)
rsi = rsi_wilder(close, period=14)

# 크로스 감지
cross_up = (ema7 > ema100) and (prev_ema7 <= prev_ema100)
cross_dn = (ema7 < ema100) and (prev_ema7 >= prev_ema100)

# 진입 판단
if cross_up and adx >= 30 and 30 <= rsi <= 58:
    enter_long(size=balance * 0.20 * 10)
    set_stop_loss(entry_price * 0.91)  # -9%

if cross_dn and adx >= 30 and 30 <= rsi <= 58:
    enter_short(size=balance * 0.20 * 10)
    set_stop_loss(entry_price * 1.09)  # +9%
```

### 12-2. 청산 로직

```python
# 매 5분봉마다 체크
pnl_pct = direction * (current_price - entry_price) / entry_price

# 1. 트레일링 추적
if pnl_pct > peak_pnl:
    peak_pnl = pnl_pct
    if peak_pnl >= 0.08:    # +8% 도달 시 트레일링 활성화
        trailing_active = True
if trailing_active:
    trailing_sl = max(trailing_sl, peak_pnl - 0.06)  # 최고점 -6%
    if pnl_pct <= trailing_sl:
        close_position("TSL")

# 2. 기본 손절 (보험용)
if pnl_pct <= -0.09:
    close_position("SL")

# 3. 반대 신호 (ADX+RSI 필터 필수)
if holding_long and short_signal_with_filter:
    close_long(); enter_short()
if holding_short and long_signal_with_filter:
    close_short(); enter_long()
```
